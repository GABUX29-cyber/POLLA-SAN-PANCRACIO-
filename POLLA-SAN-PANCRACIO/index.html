document.addEventListener('DOMContentLoaded', () => { // --- VARIABLES Y ESTADO --- let participantes = []; let resultadosDiarios = []; let finanzas = { ventas: 0, recaudado: 0.00, acumulado1: 0.00, acumulado2: 0.00, modalidad: '1_premio' }; // --- ELEMENTOS DEL DOM --- const tablaBody = document.getElementById('ranking-body'); const displayResultados = document.getElementById('numeros-ganadores-display'); const inputFiltro = document.getElementById('filtroParticipantes'); // Elementos de Stats const elVentas = document.getElementById('ventas'); const elRecaudado = document.getElementById('recaudado'); const elAcumulado1 = document.getElementById('acumulado1'); const elAcumulado2 = document.getElementById('acumulado2'); // Elementos de Premios const elPrimerPremio = document.getElementById('primer-premio'); const elSegundoPremio = document.getElementById('segundo-premio'); const elMontoDomingo = document.getElementById('monto-domingo'); const elTotalGanadores = document.getElementById('total-ganadores'); // --- 1. CARGA DE DATOS DESDE SUPABASE --- async function cargarDatos() { try { // Obtener Participantes const { data: pData, error: pError } = await _supabase .from('participantes') .select('*') .order('nro', { ascending: true }); // Obtener Resultados const { data: rData, error: rError } = await _supabase .from('resultados') .select('*') .order('id', { ascending: true }); // Obtener Finanzas const { data: fData, error: fError } = await _supabase .from('finanzas') .select('*') .single(); if (pError || rError || fError) throw new Error("Error al obtener datos"); participantes = pData || []; resultadosDiarios = rData.map(r => r.numero) || []; finanzas = fData || finanzas; renderizarResultados(rData); procesarYRenderizarTabla(); actualizarDashboard(); } catch (err) { console.error("Error en la conexión:", err); displayResultados.innerHTML = `

Error de conexión con el servidor.

`; } } // --- 2. RENDERIZADO DE RESULTADOS (CÍRCULOS) --- function renderizarResultados(lista) { if (lista.length === 0) { displayResultados.innerHTML = `

No hay resultados registrados hoy.

`; return; } displayResultados.innerHTML = ''; lista.forEach(res => { const contenedor = document.createElement('div'); contenedor.className = 'resultado-item'; contenedor.innerHTML = ` ${res.sorteo}
${res.numero}
`; displayResultados.appendChild(contenedor); }); } // --- 3. LÓGICA DE CÁLCULO Y TABLA --- function procesarYRenderizarTabla() { const busqueda = inputFiltro.value.toLowerCase(); tablaBody.innerHTML = ''; let conteoGanadores = 0; // Filtrar participantes según el buscador const filtrados = participantes.filter(p => p.nombre.toLowerCase().includes(busqueda) || p.refe.toString().includes(busqueda) ); filtrados.forEach(p => { // Cálculo de aciertos (comparando jugadas con resultados diarios) let aciertos = p.jugadas.filter(num => resultadosDiarios.includes(num)).length; if (aciertos >= 5) conteoGanadores++; const tr = document.createElement('tr'); // Determinar clase de resaltado según aciertos if (aciertos === 6) tr.className = 'ganador-max'; else if (aciertos === 5) tr.className = 'ganador-medio'; // Construir celdas de la jugada (6 columnas) let celdasJugada = ''; for (let i = 0; i < 6; i++) { let num = p.jugadas[i] || '-'; let claseAcierto = resultadosDiarios.includes(num) ? 'acertado' : ''; celdasJugada += `${num}`; } tr.innerHTML = ` ${p.nro} ${p.nombre} ${p.refe} ${celdasJugada} ${aciertos} `; tablaBody.appendChild(tr); }); elTotalGanadores.textContent = conteoGanadores; } // --- 4. ACTUALIZACIÓN DEL DASHBOARD FINANCIERO --- function actualizarDashboard() { elVentas.textContent = participantes.length; elRecaudado.textContent = `${finanzas.recaudado.toFixed(2)} BS`; elAcumulado1.textContent = `${finanzas.acumulado1.toFixed(2)} BS`; elAcumulado2.textContent = `${(finanzas.acumulado2 || 0).toFixed(2)} BS`; // Cálculo del 5% Domingo const montoDomingo = (finanzas.recaudado * 0.05).toFixed(2); elMontoDomingo.textContent = `${montoDomingo} BS`; // Lógica de Reparto de Premios (75% del recaudado tras apartar casa y domingo) const totalPremios = finanzas.recaudado * 0.75; let primerPremio = 0; let segundoPremio = 0; if (finanzas.modalidad === '2_premios') { // Reparto proporcional si hay dos premios primerPremio = totalPremios * 0.70; segundoPremio = totalPremios * 0.30; } else { // Todo al primer premio si la modalidad es normal (tarde) primerPremio = totalPremios; segundoPremio = 0.00; } elPrimerPremio.textContent = `${primerPremio.toFixed(2)} BS`; elSegundoPremio.textContent = `${segundoPremio.toFixed(2)} BS`; } // --- 5. EVENTOS --- // Buscador en tiempo real inputFiltro.addEventListener('input', procesarYRenderizarTabla); // Formatear Fecha Actual en el Header const fechaHeader = document.querySelector('header p'); const opcionesFecha = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' }; fechaHeader.innerHTML = ` ${new Date().toLocaleDateString('es-ES', opcionesFecha)}`; // Botón de Descarga PDF document.getElementById('btn-descargar-pdf').addEventListener('click', () => { window.print(); }); // --- INICIALIZACIÓN --- cargarDatos(); // Recarga automática cada 30 segundos setInterval(cargarDatos, 30000); });
